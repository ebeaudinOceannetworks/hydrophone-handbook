ONC Hydrophone Network

Ocean Networks Canada operates a massive spatial array of underwater acoustics monitoring across the country. Each hydrophone has a designated Device Code (ex: ICLISTENHF6016) and is located at a designated Location code (ex: CSQH.H1). The Location Code is used by the database, but they also have a Colloquial Name (ex: Clayoquot Slope), used by humans. In addition, there are two observatories VENUS and NEPTUNE, which regroup several sites.

Most locations have a single hydrophone, while some sites have hydrophone arrays, which constitutes several hydrophones mounted together on a platform. The hydrophone arrays naming convention has .H1, .H2, etc., added to the location code, to denote the position of the hydrophone within the array.

Main locations

  • Pacific: NEPTUNE (offshore) and VENUS (coastal) observatories.
  • Arctic: Cambridge Bay, Gascoyne Inlet, NU.
  • Atlantic: Conception Bay, Holyrood, NL.

Map

L = require("leaflet@1.9.4")
html`<link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css">`
container = html`<div style="height: 600px; border-radius: 10px; border: 1px solid #ddd;"></div>`
yield container;
{
  const map = L.map(container).setView([58, -95], 3);

  L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/Ocean/World_Ocean_Base/MapServer/tile/{z}/{y}/{x}', {
    attribution: 'Esri & Sources',
    maxZoom: 13
  }).addTo(map);

  // 1. Initialize Layer Groups
  const cableLayer = L.layerGroup().addTo(map);
  const inactiveLayer = L.layerGroup();
  const activeLayer = L.layerGroup().addTo(map);
  

  // 2. Add Cable Data to its group
  if (cable_data) {
    L.geoJSON(cable_data, {
      style: {
        color: 'grey',
        weight: 1,
        opacity: 0.7
      }
    }).addTo(cableLayer);
  }

  // 3. Loop and assign markers to groups
  status_data.locationCode.forEach((code, i) => {
    const isActive = status_data.any_active[i];
    const color = isActive ? "#28a745" : "#6c757d";
    const cleanName = status_data.locationName[i].replace(/Hydrophone [A-Z]/g, "Hydrophone Array");

    const deviceList = status_data.deviceCode[i].map((dev, idx) => {
      const status = status_data.end[i][idx] ? 
        '<span style="color:green;">Active</span>' : 
        '<span style="color:red;">Inactive</span>';
      return `<li><b>${dev}</b>: ${status}</li>`;
    }).join("");

    const marker = L.circleMarker([status_data.lat[i], status_data.lon[i]], {
      radius: 8,
      fillColor: color,
      color: "white",
      weight: 2,
      fillOpacity: 0.9
    })
    .bindPopup(`
      <div style="font-family: sans-serif; min-width: 200px;">
        <b style="font-size: 1.1em;">${cleanName}</b><br>
        <span style="color: #666; font-size: 0.9em;">${code}</span>
        <hr style="margin: 5px 0;">
        <p style="margin: 5px 0 0 0;"><b>Depth:</b> ${status_data.depth[i]} m</p>
        <p style="margin: 5px 0 0 0;"><b>Instruments at this site:</b></p>
        <ul style="margin: 5px 0; padding-left: 20px; font-size: 0.9em;">
          ${deviceList}
        </ul>
      </div>
    `);

    // Add marker to the appropriate group
    if (isActive) {
      marker.addTo(activeLayer);
    } else {
      marker.addTo(inactiveLayer);
    }
  });

  // 4. Create the Overlay Toggle Menu
  const overlays = {
    "<span style='color: #28a745; font-weight: bold;'>● Active Sites</span>": activeLayer,
    "<span style='color: #6c757d; font-weight: bold;'>○ Inactive Sites</span>": inactiveLayer,
    "Cables": cableLayer
  };

  // Add the control box to the top right
  L.control.layers(null, overlays, { collapsed: false }).addTo(map);
}
