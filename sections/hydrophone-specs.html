Hydrophone Specs & Manufacturers

Most hydrophones at ONC are from Ocean Sonics: icListen HF (10 Hz - 200 kHz; high frequency), and icListen AF (10 Hz - 12 kHz; low-frequency). A few arrays use Geospectrum hydrophones provided by JASCO.

all_cols = {
  if (specs_data && specs_data.length > 0) {
    return Object.keys(specs_data[0]);
  } else {
    return [];
  }
}

// 2. Define our labels based on the safe extraction
label_col = all_cols.length > 0 ? all_cols[0] : ""
model_cols = all_cols.length > 0 ? all_cols.slice(1) : []

// NEW: Extract the "Device" row (first row) and separate it from the rest of the specs
deviceRow = specs_data.length > 0 ? specs_data[0] : {}
specRows = specs_data.length > 1 ? specs_data.slice(1) : []

// 3. Map YOUR specific manufacturers to colors
brandColors = ({
  "JASCO": "#ffc107",        // Yellow
  "Ocean Sonics": "#03a9f4", // Blue
  "Naxys": "#9c27b0",        // Purple
  "Feature": "#e64a19"       // Orange (for the Manufacturer label column)
})

// 4. Create the interactive checkbox UI for selecting models
viewof selectedModels = Inputs.checkbox(
  model_cols, 
  { 
    label: html`<b>Select Models to Compare:</b>`, 
    value: model_cols.slice(0, 3), // Select first 3 by default
    // Show the Device name in the checkboxes, fallback to Manufacturer if missing
    format: x => deviceRow[x] && deviceRow[x] !== "--" ? deviceRow[x] : x.split('.')[0]
  }
)

// 5. Render the styled table!
html`
<style>
  .table-responsive {
    overflow-x: auto; 
    overflow-y: auto;
    max-height: 600px; /* Enables vertical scrolling within the table area */
    border-radius: 12px; /* Smooth corners for the container */

  .branded-table {
    border-collapse: separate;
    border-spacing: 8px 0;
    width: 100%;
    margin-top: 10px;
    font-family: sans-serif;
  }
  .branded-table th {
    padding: 20px 10px;
    color: white;
    font-weight: bold;
    border-radius: 12px 12px 0 0;
    text-align: center;
    border: none;
    min-width: 150px; 
    position: sticky; /* Makes the header sticky */
    top: 0;           /* Pins it to the top */
    z-index: 10;      /* Ensures it floats above the table body */
    box-shadow: 0px 4px 6px -4px rgba(0,0,0,0.4); /* Slight shadow to separate from content */
  }
  .branded-table td {
    padding: 15px 10px;
    text-align: center;
    border: none;
    background-color: #f8f9fa;
    color: #333;
    min-width: 150px; 
  }
  .branded-table tr:nth-child(even) td {
    background-color: #ffffff;

  .footer-row td {

  <table class="branded-table">
    <thead>
      <tr>
        <th style="background-color: ${brandColors.Feature || '#e64a19'}; min-width: 220px;">Device / Model</th>
        ${selectedModels.map(m => {
          // Use the actual device name for the header
          let deviceName = deviceRow[m] && deviceRow[m] !== "--" ? deviceRow[m] : m.split('.')[0];

          // Determine color based on manufacturer
          let color = "#333";
          if (m.includes("JASCO")) color = brandColors.JASCO;
          if (m.includes("Ocean")) color = brandColors.OceanSonics;
          if (m.includes("Naxys")) color = brandColors.Naxys;

          return html`<th style="background-color: ${color}">${deviceName}</th>`;
        })}
      </tr>
    </thead>

      <tr>

        ${selectedModels.map(m => {

        })}
      </tr>

      <!-- Render the rest of the specifications -->
      ${specRows.map(row => html`
        <tr>
          <td style="font-weight: bold; text-align: left; color: #333;">${row[label_col]}</td>
          ${selectedModels.map(m => html`<td>${row[m] || "--"}</td>`)}
        </tr>
      `)}
      
      <!-- Branded Footer -->
      <tr class="footer-row">
        <td style="background-color: ${brandColors.Feature || '#e64a19'}"></td>
        ${selectedModels.map(m => {
          let color = "#333";
          if (m.includes("JASCO")) color = brandColors.JASCO;

          if (m.includes("Naxys")) color = brandColors.Naxys;
          return html`<td style="background-color: ${color}"></td>`;
        })}
      </tr>
    </tbody>
  </table>
</div>
